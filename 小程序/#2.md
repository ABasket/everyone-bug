count_down: function() {
    let that = this

    // 获取后台倒计时时间
    let s = that.data.participation.uorder.countdown

    let min = s % 3600;
    let H = Math.floor(s / 3600);
    let M = Math.floor(min / 60);
    let S = Math.floor(s % 60);

    // 开启定时器
    let count_down = setInterval(() => {
      // 后端结束则前台页结束或者已经购买则计时器页结束
      if (that.data.participation.uorder.state === 2 || that.data.participation.uorder.state === 4) {
        clearInterval(that.data.count_down)
        S = '0' + 0;
        M = '0' + 0;
        H = '0' + 0;
        that.setData({
          H,
          M,
          S
        })
      };
      if (S > 0 || M > 0 || H > 0) {
        S -= 1
      }
      // 开启补零操作
      that.get_zero(S, H, M);

      if (S <= 0 && M > 0) {
        M -= 1
        S = 60
      }
      if (M <= 0 && H > 0) {
        H -= 1
        M = 59
      }
    }, 1000);
    that.setData({
      count_down
    })
  },

  // 补零操作
  get_zero: function(S, H, M) {
    if (S < 10) {
      S = '0' + S
    }

    if (M < 10) {
      M = '0' + M
    }

    if (H < 10) {
      H = '0' + H
    }
    // 前端计算器都为零的时候全部为0
    if (S <= 0 && M <= 0 && H <= 0) {
      clearInterval(this.data.count_down)
      S = '0' + 0;
      M = '0' + 0;
      H = '0' + 0;
      this.setData({
        H,
        M,
        S
      })
    };
    this.setData({
      H,
      M,
      S
    })
  },```
